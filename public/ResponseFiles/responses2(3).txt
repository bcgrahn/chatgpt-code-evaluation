
def triple_and(a_arg, b_arg, c_arg):
    """Returns True only if all parameters are True and False otherwise.

    Args:
        a_arg: First boolean parameter.
        b_arg: Second boolean parameter.
        c_arg: Third boolean parameter.

    Returns:
        bool: True only if all parameters are True and False otherwise.
    """
    return a_arg and b_arg and c_arg
